  CREATE TABLE GENEROS

  CREATE TABLE NACIONALIDAD
   (  IDNACIONALIDAD NUMBER(11,0) NOT NULL,

  PRECIO NUMBER(11,0) DEFAULT 0

   )  ;

  ACTORES CLOB,
