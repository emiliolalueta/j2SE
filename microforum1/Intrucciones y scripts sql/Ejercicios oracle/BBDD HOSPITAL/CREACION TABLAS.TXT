	(DEPT_NO NUMBER(2,0)

);

CREATE TABLE EMP
	(EMP_NO NUMBER(4,0)
	,APELLIDO VARCHAR2(19)
	,OFICIO VARCHAR2(10)
	,DIR NUMBER(4,0)
	,FECHA_ALT DATE
	,SALARIO NUMBER(10,0)
	,COMISION NUMBER(10,0)

);

CREATE TABLE HOSPITAL(
	HOSPITAL_COD NUMBER(2)
	,NOMBRE VARCHAR2(19)
	,DIRECCION VARCHAR2(20),
	TELEFONO VARCHAR2(9),
	NUM_CAMA NUMBER(3)
);

CREATE TABLE SALA(
	HOSPITAL_COD NUMBER(2)
	,SALA_COD NUMBER(2)
	,NOMBRE VARCHAR2(20)
	,NUM_CAMA NUMBER(3)
);

CREATE TABLE DOCTOR(
	HOSPITAL_COD NUMBER(2)

	,APELLIDO VARCHAR2(20)
	,ESPECIALIDAD VARCHAR(14),
	SALARIO NUMBER(10)
);

CREATE TABLE PLANTILLA(
	HOSPITAL_COD NUMBER(2)
	,SALA_COD NUMBER(2)

	,APELLIDO VARCHAR2(15)
	,FUNCION VARCHAR2(10)

);

CREATE TABLE ENFERMO(
